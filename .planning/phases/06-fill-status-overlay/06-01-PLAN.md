Replace the 3-line stub with a complete IIFE. Implement exactly in this order:

1. IIFE wrapper + namespace guard:
   (function() { 'use strict'; window.JobFill = window.JobFill || {}; ... })();

2. Platform detection (at init time):
   const hostname = window.location.hostname;
   const platforms = window.JobFill.platforms || {};
   let _platform = Object.values(platforms).find(function(p) { return p.matches(hostname); }) || null;

3. safeRuntimeCall(fn) helper — wraps any chrome.runtime call in try/catch:
   - If err.message includes 'Extension context invalidated': call window.JobFill.overlay.showBanner('JobFill was updated — please reload this tab.')
   - Otherwise: console.warn('[JobFill]', err.message)
   - Always returns null on catch

4. runFill() async function:
   a. Check window.JobFill.filler.startFill() — if returns false, return { error: 'fill_in_progress' }
   b. try block:
      - const profile = await window.JobFill.storage.getProfile()
      - if (!profile) return { error: 'no_profile' } (endFill in finally)
      - if _platform exists: const answerBank = await window.JobFill.storage.getAnswerBank(); results = await _platform.fill(profile, answerBank)
      - else: results = [{ field: 'Platform', status: 'skipped', reason: 'No platform module loaded for ' + hostname }]
      - await window.JobFill.storage.saveFillStatus(window._jobfillTabId || null, results)
      - window.JobFill.overlay.showResults(results)
      - return { ok: true, count: results.length }
   c. catch(err): return { error: err.message }
   d. finally: window.JobFill.filler.endFill()

5. onMessage listener (NON-ASYNC — this is mandatory):
   chrome.runtime.onMessage.addListener(function(msg, _sender, sendResponse) {
     // Wrap in safeRuntimeCall is not applicable here — the listener itself must not throw
     // Instead use try/catch inside the body
     try {
       if (msg.type === 'DO_FILL') {
         // Capture tabId from message for saveFillStatus
         if (msg.tabId) window._jobfillTabId = msg.tabId;
         if (window.JobFill.filler.isFilling()) {
           sendResponse({ error: 'fill_in_progress' });
           return true;
         }
         runFill().then(sendResponse).catch(function(err) {
           sendResponse({ error: err.message });
         });
         return true; // MANDATORY — keeps async channel open
       }
     } catch (err) {
       if (err.message && err.message.includes('Extension context invalidated')) {
         window.JobFill.overlay.showBanner('JobFill was updated — please reload this tab.');
       } else {
         console.warn('[JobFill content]', err.message);
       }
     }
   });

6. MutationObserver for SPA navigation:
   const _observer = new MutationObserver(function(mutations) {
     // NFR-3.1: NO storage reads inside this callback
     const hasNewInputs = mutations.some(function(m) {
       return Array.from(m.addedNodes).some(function(n) {
         if (n.nodeType !== 1) return false;
         // Guard: ignore own overlay DOM mutations
         if (n.closest && n.closest('#jobfill-overlay-host')) return false;
         return n.matches('input,select,textarea') || !!n.querySelector('input,select,textarea');
       });
     });
     // Only show button if not already showing (overlay host absent)
     if (hasNewInputs && !document.getElementById('jobfill-overlay-host')) {
       window.JobFill.overlay.showButton(runFill);
     }
   });
   _observer.observe(document.body, { childList: true, subtree: true });
   window.JobFill.overlay.setObserverRef(_observer);

7. Initial render:
   window.JobFill.overlay.showButton(runFill);

- MutationObserver callback MUST NOT call chrome.storage.* or window.JobFill.storage.*
- n.closest('#jobfill-overlay-host') check MUST be present before input detection
- Platform detection guards with `window.JobFill.platforms || {}`
