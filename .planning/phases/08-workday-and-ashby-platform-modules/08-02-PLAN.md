Key Workday differences vs greenhouse.js:
1. resolveField uses shadowQuery (NOT document.querySelector):
   window.JobFill.filler.shadowQuery(document.body, selector)

   function isVisible(el) { return el.offsetParent !== null; }

   var ok = window.JobFill.filler.fillField(el, value);
   if (ok) window.JobFill.events.dispatchBlur(el);
